\documentclass{article}\usepackage{amsmath}\begin{document}\title{Simplex Solver}\maketitle\begin{flushleft}\textbf{Problem}\end{flushleft}\begin{flushleft}Given the following linear system and objective function, find the optimal solution.\end{flushleft}\begin{equation*}\max{ x_1 + x_2 + x_3 + x_4 } \\ \end{equation*}\[\left\{\begin{array}{c} 4x_1 + 3x_2 + 4x_3 + 2x_4 \leq 1 \\  3x_1 + 4x_2 + 6x_3 + 5x_4 \leq 1 \\  2x_1 + 5x_2 + x_3 + 3x_4 \leq 1 \\ \end{array}\right.\]\begin{flushleft}\textbf{Solution}\end{flushleft}\begin{flushleft}Add slack variables to turn all inequalities to equalities.\end{flushleft}\[\left\{\begin{array}{c} 4x_1 + 3x_2 + 4x_3 + 2x_4 + s_1 = 1 \\  3x_1 + 4x_2 + 6x_3 + 5x_4 + s_2 = 1 \\  2x_1 + 5x_2 + x_3 + 3x_4 + s_3 = 1 \\ \end{array}\right.\]\begin{flushleft}Create the initial tableau of the new linear system.\end{flushleft}\begin{equation*}\begin{bmatrix}\begin{array}{ccccccc|c}x_1 &x_2 &x_3 &x_4 &s_1 &s_2 &s_3 &b \\ \hline4 & 3 & 4 & 2 & 1 & 0 & 0 & 1 \\3 & 4 & 6 & 5 & 0 & 1 & 0 & 1 \\2 & 5 & 1 & 3 & 0 & 0 & 1 & 1 \\ \hline-1 & -1 & -1 & -1 & 0 & 0 & 0 & 0 \\\end{array}\end{bmatrix}\begin{array}{c}\\s_1 \\s_2 \\s_3 \\\\\end{array}\end{equation*}\begin{flushleft}There are negative elements in the bottom row, so the current solution is not optimal. Thus, pivot to improve the current solution. The entering variable is $x_1$ and the departing variable is $s_1$.\end{flushleft}\begin{flushleft}Perform elementary row operations until the pivot element is 1 and all other elements in the entering column are 0.\end{flushleft}\begin{equation*}\begin{bmatrix}\begin{array}{ccccccc|c}x_1 &x_2 &x_3 &x_4 &s_1 &s_2 &s_3 &b \\ \hline1 & 3/4 & 1 & 1/2 & 1/4 & 0 & 0 & 1/4 \\0 & 7/4 & 3 & 7/2 & -3/4 & 1 & 0 & 1/4 \\0 & 7/2 & -1 & 2 & -1/2 & 0 & 1 & 1/2 \\ \hline0 & -1/4 & 0 & -1/2 & 1/4 & 0 & 0 & 1/4 \\\end{array}\end{bmatrix}\begin{array}{c}\\x_1 \\s_2 \\s_3 \\\\\end{array}\end{equation*}\begin{flushleft}There are negative elements in the bottom row, so the current solution is not optimal. Thus, pivot to improve the current solution. The entering variable is $x_4$ and the departing variable is $s_2$.\end{flushleft}\begin{flushleft}Perform elementary row operations until the pivot element is 1 and all other elements in the entering column are 0.\end{flushleft}\begin{equation*}\begin{bmatrix}\begin{array}{ccccccc|c}x_1 &x_2 &x_3 &x_4 &s_1 &s_2 &s_3 &b \\ \hline1 & 1/2 & 4/7 & 0 & 5/14 & -1/7 & 0 & 3/14 \\0 & 1/2 & 6/7 & 1 & -3/14 & 2/7 & 0 & 1/14 \\0 & 5/2 & -19/7 & 0 & -1/14 & -4/7 & 1 & 5/14 \\ \hline0 & 0 & 3/7 & 0 & 1/7 & 1/7 & 0 & 2/7 \\\end{array}\end{bmatrix}\begin{array}{c}\\x_1 \\x_4 \\s_3 \\\\\end{array}\end{equation*}\begin{flushleft}There are no negative elements in the bottom row, so we know the solution is optimal. Thus, the solution is: \end{flushleft}\begin{equation*}s_1 = 0, s_2 = 0, s_3 = \frac{5}{14}, x_1 = \frac{3}{14}, x_2 = 0, x_3 = 0, x_4 = \frac{1}{14}, z = \frac{2}{7}\end{equation*}\end{document}
